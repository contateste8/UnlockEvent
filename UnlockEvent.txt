local function createSafetyPlatform(position)

table.insert(platforms, part)
end

local hasCandy = false

    task.wait(1.5)
    for _, obj in ipairs(workspace:GetDescendants()) do
        if obj:IsA("BasePart") and string.find(obj.Name, "CandyCorn") then
            hasCandy = true
            break
        end
    end

    label.Text = "Nenhum doce encontrado"

    if obj:IsA("BasePart") and string.find(obj.Name, "CandyCorn") then
        table.insert(candyList, obj)
    end
end

end

end
end
